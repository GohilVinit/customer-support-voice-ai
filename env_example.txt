# Enhanced Customer Support Call Analysis System - Environment Variables
# Copy this file to .env and fill in your API keys

# =============================================================================
# REQUIRED API KEYS
# =============================================================================

# OpenAI API Key - Required for speech-to-text transcription
# Get from: https://platform.openai.com/api-keys
OPENAI_API_KEY=your_openai_api_key_here

# Google Gemini API Key - Required for chat/completion and insights extraction
# Get from: https://makersuite.google.com/app/apikey
GEMINI_API_KEY=your_gemini_api_key_here

# =============================================================================
# OPTIONAL API KEYS
# =============================================================================

# Hugging Face Token - Optional for speaker diarization
# Get from: https://huggingface.co/settings/tokens
# Required for: Speaker separation in customer support calls
HF_TOKEN=your_huggingface_token_here

# MeetingBaas API Key - Optional for meeting recording
# Get from: https://meetingbaas.com/
# Required for: Live meeting recording and transcription
MEETINGBAAS_API_KEY=your_meetingbaas_api_key_here

# =============================================================================
# GOOGLE DRIVE INTEGRATION (OPTIONAL)
# =============================================================================

# Google Drive credentials file path
# Download from: https://console.cloud.google.com/apis/credentials
GOOGLE_APPLICATION_CREDENTIALS=path_to_your_credentials.json

# =============================================================================
# SYSTEM CONFIGURATION
# =============================================================================

# Memory management threshold (percentage)
# Default: 80 (for 4GB RAM systems)
MEMORY_THRESHOLD=80

# Upload folder for temporary files
UPLOAD_FOLDER=./uploads

# Vector store path
VECTORSTORE_PATH=./unified_vectorstore

# =============================================================================
# ANALYSIS CONFIGURATION
# =============================================================================

# Enable/disable analysis components (true/false)
ENABLE_DIARIZATION=true
ENABLE_INSIGHTS=true
ENABLE_SUMMARY=true
ENABLE_EVALUATION=true
ENABLE_AGENT_ANALYSIS=true

# =============================================================================
# LOGGING CONFIGURATION
# =============================================================================

# Log level (DEBUG, INFO, WARNING, ERROR)
LOG_LEVEL=INFO

# =============================================================================
# SERVER CONFIGURATION
# =============================================================================

# Flask server host
HOST=0.0.0.0

# Flask server port
PORT=5000

# Debug mode (true/false)
DEBUG=false
